# FAQs as a Service (faqaas)


## DB Setup

```
psql -U postgres
\c faqaas
```

```sql
CREATE TABLE faqs (
  id SERIAL PRIMARY KEY,
  question TEXT,
  answer TEXT
);

CREATE TABLE faq_texts (
  id SERIAL PRIMARY KEY,
  faq_id INTEGER REFERENCES faqs (id),
  locale TEXT,
  question TEXT,
  answer TEXT,
  CONSTRAINT texts_faq_id_locale unique(faq_id,locale)
);

CREATE MATERIALIZED VIEW search_index AS
```
